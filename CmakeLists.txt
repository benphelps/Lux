cmake_minimum_required(VERSION 3.10)

project(phelt)

# set include directory
include_directories(src/include)

add_executable(phelt
    src/common.c
    src/main.c
    src/debug.c
    src/chunk.c
    src/value.c
    src/memory.c
    src/vm.c
    src/compiler.c
    src/scanner.c
    src/object.c
    src/table.c
    src/string.c
    src/native/system.c
    src/native/math.c
    src/native/file.c
    src/native/http.c
    src/native/array.c
    src/native/string.c
    src/native/table.c
    src/native/native.c
)

target_link_libraries(phelt curl)
target_compile_options(phelt PRIVATE -Wall -Wextra -Wpedantic -Werror -Wno-gnu-label-as-value -Wno-gnu-zero-variadic-macro-arguments)
